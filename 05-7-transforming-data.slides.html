In [3]:
# A histogram of the percentage of unemployed in all US counties.
fig, ax = plt.subplots()

sns.histplot(data=df, x= "unemployment_rate", palette=colors_blue)

ax.xaxis.set_major_formatter(ticker.EngFormatter('%'))
plt.xlabel("Unemployment rate")
sns.despine();
